CREATE DATABASE military_db;

USE military_db;

CREATE TABLE soldiers (
  id INT AUTO_INCREMENT PRIMARY KEY,
  name VARCHAR(100),
  age INT,
  gender ENUM('Male','Female'),
  rank VARCHAR(50),
  unit VARCHAR(100),
  enlist_date DATE,
  contact VARCHAR(100)
);


CREATE TABLE users (
  id INT AUTO_INCREMENT PRIMARY KEY,
  username VARCHAR(50) UNIQUE NOT NULL,
  password VARCHAR(255) NOT NULL
);

ALTER TABLE soldiers ADD soldier_code VARCHAR(20) UNIQUE;


ALTER TABLE soldiers ADD COLUMN photo VARCHAR(255) NULL;
